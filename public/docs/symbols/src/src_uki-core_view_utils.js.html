  6 uki.view.utils = new function() {
  7     var proto = this;
  8     
  9     /** @exports proto as uki.view.utils */
 10     /**#@+ @memberOf uki.view.utils */
 11     
 12     proto.visibleRect = function (from, upTo) {
 13         var queue = [],
 14             rect, i, tmpRect, c = from;
 15             
 16         do {
 17             queue[queue.length] = c;
 18             c = c.parent();
 19         } while (c && c != upTo);
 20         
 21         if (upTo && upTo != from) queue[queue.length] = upTo;
 22 
 23         for (i = queue.length - 1; i >= 0; i--){
 24             c = queue[i];
 25             tmpRect = visibleRect(c);
 26             rect = rect ? rect.intersection(tmpRect) : tmpRect;
 27             rect.x -= c.rect().x;
 28             rect.y -= c.rect().y;
 29             
 30         };
 31         return rect;
 32     };
 33     
 34     proto.top = function(c) {
 35         while (c.parent()) c = c.parent();
 36         return c;
 37     };
 38     
 39     proto.offset = function(c, upTo) {
 40         var offset = new Point(),
 41             rect;
 42         
 43         while (c && c != upTo) {
 44             rect = c.rect();
 45             offset.x += rect.x;
 46             offset.y += rect.y;
 47             if (c.scrollTop) {
 48                 offset.x -= c.scrollLeft();
 49                 offset.y -= c.scrollTop();
 50             }
 51             c = c.parent();
 52         }
 53         return offset;
 54     };
 55     
 56     proto.scrollableParent = function(c) {
 57         do {
 58             if (uki.isFunction(c.scrollTop)) return c;
 59             c = c.parent();
 60         } while (c);
 61         return null;
 62     };
 63     
 64     /** @inner */
 65     function visibleRect (c) {
 66         return c.visibleRect ? c.visibleRect() : c.rect().clone();
 67     }
 68     /**#@-*/ 
 69 };
